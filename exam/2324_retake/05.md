SuperSparseMatrix <-- function(nrow, ncol, i, j, v) {
  # 参数检查
  assertCount(nrow, tol = 0)
  assertCount(ncol, tol = 0)
  assertCount(i, tol = 0)
  assertCount(j, tol = 0)
  assertNumber(v, na.ok = TRUE)
  
  # 边界检查 - 自定义错误信息
  if (i > nrow | j > ncol) {
    stop("entry indices out of bounds")
  }
  
  # 创建 S3 对象
  # `structure()` 给列表添加 `class` 属性，使其成为 S3 对象
  structure(list(
    nrow = nrow,
    ncol = ncol,
    i = i,
    j = j, 
    v = v
  ), class = "SuperSparseMatrix")
}

# 打印方法
print.SuperSparseMatrix <- function(x, ...) {
  cat(sprintf("A sparse %sx%s matrix with value %s at [%s, %s].\n", 
              x$nrow, x$ncol, x$i, x$j, x$v))
  invisible(x)
}

# 维度方法 
# 返回 `c(行数, 列数)`，定义了 `dim()` 后，`nrow()` 和 `ncol()` 自动可用
dim.SuperSparseMatrix <- function(x) {
  c(x$nrow, x$ncol)
}

# 转换方法
as.matrix.SuperSparseMatrix <- function(x, ...) {
  mat <- matrix(0, nrow = x$nrow, ncol = x$ncol)  # 创建全 0 矩阵
  mat[x$i, x$j] <- x$v  # 设置非零值
  mat
}
