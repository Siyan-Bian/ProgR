**根据 DataTable 的索引和值给矩阵同一索引位置赋值**
```r
sparseToMatrix <- function(n.row, n.col, entries) {
  checkmate::assert_number(n.row, lower = 1)
  checkmate::assert_number(n.col, lower = 1)
  checkmate::assert_data_table(entries)
  checkmate::assert_names(names(entries), must.include = c("i", "j", "value"))
  
  result <- matrix(0, nrow = n.row, ncol = n.col)
  
  if (anyDuplicated(entries[, .(i,  j)])) {
    stop("entries contain duplicate indices")
  }
  
  if (any(entries$i < 1 | entries$i > n.row | entries$j < 1 | entries$j > n.col)) {
    stop("indices out of bounds")
  }
  
  if (nrow(entries) > 0) {
    result[cbind(entries$i, entries$j)] <- entries$value
  }
  
  result
}
```
