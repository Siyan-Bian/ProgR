取每个公司薪资第二高的员工，若每个公司仅有一名员工，则返回空的datatable
```r
secondHighestEarner <- function(employees) {
  # 输入验证
  checkmate::assert_data_table(employees)
  checkmate::assert_names(names(employees), must.include = c("name", "company", "salary"))
  
  # 处理空表
  if (nrow(employees) == 0) {
    return(data.table(company = character(0), name = character(0), salary = numeric(0)))
  }
  
  # 按公司分组，按薪资降序排列，取第二高
  result <- employees[order(-salary), .SD[2], by = company]
  
  # 移除只有一个员工的公司（第二高是 NA）
  # 当公司只有一个员工时，`.SD[2]` 取第二行，但第二行不存在，返回 NA
  result <- result[!is.na(name)]
  
  # 调整列顺序
  result[, .(company, name, salary)]
}
```
