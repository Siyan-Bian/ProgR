```r
StackContainer <- R6Class("StackContainer",
  # your code
  private <- list(
    stack <- character(0)
  ),
  
  public <- list(
    # 添加到栈顶
    push <- function(item) {
      checkmate::assert_character(item, len = 1)
      private$stack <- c(private$stack, item)  # 添加到末尾
      invisible(self)  # 返回自身，支持链式调用
    },
    
    # 移除并返回栈顶
    pop <- function() {
      if (length(private$stack) == 0) {
        stop("stack is empty")
      }
      
      item <- private$stack[length(private$stack)]  # 取最后一个
      private$stack <- private$stack[-length(private$stack)]  # 删除最后一个
    },
    
    # 返回元素个数
    size <- function() {
      length(private$stack)
    }
  )
)
```
* **向量末尾代表栈顶**
