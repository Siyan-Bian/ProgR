VendingMachine <- R6Class("VendingMachine",
  # your code
  # 私有数据
  private = list(
    inventory = NULL  # 存储库存
  ),
  
  # 公开方法
  public = list(
    initialize = function(inventory) {
      checkmate::assert_numeric(inventory, lower = 0, any.missing = FALSE)
      checkmate::assert_names(names(inventory), type = "named")
      private$inventory <- inventory
    },
    
    buy = function(item) {
      checkmate::asert_string(item)
      
      if (!(item %in% names(private$inventory))) {
        stop("item not sold")
      }
      
      if (private$inventory[[item]] == 0) {
        stop("out of stock")
      }
      
      private$inventory[[item]] <- private$inventory[[item]] - 1
      invisible(self)
    },
    
    getInventory = function() {
      private$inventory
    }
  )
)
