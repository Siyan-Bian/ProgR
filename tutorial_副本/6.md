nsum <- function(n) {
  result <- 0
  for (i in seq_len(n)) {
    result <- result + i
  }
  result
}

## Ex3: 移除丢失列
```r
dropMissingCols <- function(df) {
  drop <- character(0)  #要删除的列名
  for (col in colnames(df)) {
    for (row in seq_len(nrow(df))) {
      #如果发现 NA，就把列名加入 drop
      #如果同一列发现多个 NA，使用 union 去重列名
      if (is.na(df[row, col])) drop <- union(drop, col)
    }
  }
  df[, setdiff(colnames(df), drop), drop = FALSE]
  #setdiff()集合差运算：所有列名-要删除的列名=保留的列名
}
```

## Ex4: 斐波那契数列
* 斐波那契数列定义为: `F(n) = F(n-1) + F(n-2)`
* 这里定义`x`th order fibonacci 为: `F(n, x) = F(n-1, x) + F(n-2, x) + ... + F(n - x, x)`
* `F(n, x)` is 1 for `n == 1`, and 0 for `n < 1`.
```r
hoFib <- function(n, x) {
  if (n < 1) return(0)
  if (n == 1) return(1)
  result <- 0
  for (subtract in seq_len(x)) {
    result <- result + hoFib(n - subtract, x)
  }
  return(result)
}
```

## Ex5: 检查字符是否是平衡的
```r
isBalanced <- function(string) {
  items <- strsplit(string, "")[[1]]
  stack <- character(0)
  is.balanced <- TRUE
  #遇到左括号（如 `(` 或 `[`），就把它“压入”栈顶
  for (char in items) {
    if (char == "(") {
      stack <- c(stack, char)
    }
    if (char == "[") {
      stack <- c(stack, "[")
    }
    
    if (char == ")") {
      if (length(stack) == 0) {
        is.balanced <- FALSE
      } else {
        #遇到右括号（如 `)` 或 `]`），就查看栈顶最近的一个左括号是否与其匹配
        if (tail(stack, 1) != "(") is.balanced <- FALSE
        stack <- stack[seq_len(length(stack) - 1)]
      }
    }
    if (char == "]") {
      if (length(stack) == 0) {
        is.balanced <- FALSE
      } else {
        if (tail(stack, 1) != "[") is.balanced <- FALSE
        stack <- stack[seq_len(length(stack) - 1)]
      }
    }
  }
  is.balanced
}
```
